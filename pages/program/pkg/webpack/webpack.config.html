webpack.config.js

devServer

基本配置项 

"devServer": {
    "host": "0.0.0.0",
    "port": 20000,
    "proxy": {}
}

overlay

控制是否在浏览器页面中全屏显示错误信息： 

"devServer": {
    "client": {
        "overlay": false
    }
}

proxy

比如，我们希望本地的请求 /api/users 转发到 http://www.example.tech:8080/api/users 就可以： 

"devServer": {
    "proxy": {
        "/api": {
            "target": "http://www.example.tech:8080/",

            // 表示是否更新代理后请求的headers中host地址
            "changeOrigin": true
        }
    }
}

而如果你希望转发后没有前缀api，也就是直接是： http://www.example.tech:8080/users，需要添加一个配置： 

"devServer": {
    "proxy": {
        "/api": {
            "target": "http://www.example.tech:8080/",
            "changeOrigin": true,
            "pathRewrite": { '^/api': '' },
        }
    }
}
