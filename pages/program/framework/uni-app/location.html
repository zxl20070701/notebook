获取位置信息 

// 授权
uni.authorize({
    scope: 'scope.userLocation',
    success() {

        // 获取
        uni.getLocation({
            type: 'gcj02',
            success: function(res) {
                console.log(res)
            }
        })
    }
})

配置权限

针对不同端，如果希望上面的方法调用成功，还需要在 manifest.json 中进行权限配置。

微信小程序 

"mp-weixin": {
    "permission": {
        "scope.userLocation": {
            "desc": "您的位置信息将用于小程序页面"
        }
    },
    "requiredPrivateInfos": ["getLocation"]
}
